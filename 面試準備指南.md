```javascript
// 亮點：自動合併相同商品
function addToCart(item) {
    let cart = loadCart();
    const existingIndex = cart.findIndex(cartItem => 
        cartItem.name === item.name && cartItem.size === item.size
    );
    
    if (existingIndex > -1) {
        cart[existingIndex].quantity += item.quantity;
    } else {
        cart.push(item);
    }
    saveCart(cart);
}
```
